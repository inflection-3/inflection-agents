# x402 — HTTP Micropayment Protocol

## Overview

x402 is an open protocol built on the HTTP `402 Payment Required` status code. A server responds to a request with `402` and payment details; the client pays (USDC on Base), then retries with proof of payment. This is entirely machine-native — no human interaction needed at any step.

Best for: agent-to-agent payments, API micropayments, pay-per-call services.

---

## Prerequisites

- Node.js 18+
- A wallet with testnet USDC (free from faucet — see below)
- Anthropic API key for the agent

## Building a Paid API Server

This is the service the agent will pay to access.

```typescript
// src/server.ts
import express from "express";
import { paymentMiddleware } from "x402-express";

const app = express();

// Any route listed here requires payment before responding
app.use(
  paymentMiddleware({
    facilitatorUrl: "https://x402.org/facilitator",
    routes: {
      "/api/data": {
        price: "$0.001",           // 0.001 USDC per call
        network: "base-sepolia",   // testnet
      },
      "/api/premium": {
        price: "$0.01",
        network: "base-sepolia",
      },
    },
  })
);

app.get("/api/data", (req, res) => {
  res.json({
    result: "Here is your paid data",
    timestamp: new Date().toISOString(),
    requestId: Math.random().toString(36).slice(2),
  });
});

app.get("/api/premium", (req, res) => {
  res.json({
    result: "Here is your premium paid data",
    insights: ["insight_1", "insight_2", "insight_3"],
  });
});

app.listen(process.env.PORT || 3001, () => {
  console.log(`Paid API server running on port ${process.env.PORT || 3001}`);
});
```

## Building the Agent (Client)

```typescript
// src/agent.ts
import Anthropic from "@anthropic-ai/sdk";
import axios from "axios";
import { withPaymentInterceptor } from "x402-axios";
import { createAgentWallet } from "./wallet";

const client = new Anthropic();

// Create a paying HTTP client — automatically handles 402 responses
const wallet = createAgentWallet();
const payingClient = withPaymentInterceptor(axios.create(), wallet);

const tools: Anthropic.Tool[] = [
  {
    name: "fetch_paid_data",
    description: "Fetch data from a paid API endpoint, payment is handled automatically",
    input_schema: {
      type: "object" as const,
      properties: {
        endpoint: {
          type: "string",
          description: "The API endpoint to call (e.g. /api/data or /api/premium)",
        },
      },
      required: ["endpoint"],
    },
  },
  {
    name: "check_wallet_balance",
    description: "Check the agent wallet's current USDC balance on testnet",
    input_schema: {
      type: "object" as const,
      properties: {},
      required: [],
    },
  },
];

async function handleToolCall(
  toolName: string,
  toolInput: Record<string, unknown>
): Promise<string> {
  const baseUrl = `http://localhost:${process.env.PORT || 3001}`;

  if (toolName === "fetch_paid_data") {
    const { endpoint } = toolInput as { endpoint: string };

    // payingClient intercepts the 402, pays automatically, then retries
    const response = await payingClient.get(`${baseUrl}${endpoint}`);

    return JSON.stringify({
      success: true,
      data: response.data,
      paid: true,
    });
  }

  if (toolName === "check_wallet_balance") {
    const { createPublicClient, http, formatUnits } = await import("viem");
    const { baseSepolia } = await import("viem/chains");
    const { erc20Abi } = await import("viem");

    const publicClient = createPublicClient({
      chain: baseSepolia,
      transport: http(),
    });

    // USDC contract on Base Sepolia
    const USDC_ADDRESS = "0x036CbD53842c5426634e7929541eC2318f3dCF7e";

    const { privateKeyToAccount } = await import("viem/accounts");
    const account = privateKeyToAccount(
      process.env.WALLET_PRIVATE_KEY as `0x${string}`
    );

    const balance = await publicClient.readContract({
      address: USDC_ADDRESS,
      abi: erc20Abi,
      functionName: "balanceOf",
      args: [account.address],
    });

    return JSON.stringify({
      address: account.address,
      usdc_balance: formatUnits(balance, 6),
      network: "base-sepolia (testnet)",
    });
  }

  return JSON.stringify({ error: "Unknown tool" });
}

export async function runX402Agent(task: string) {
  console.log(`\nAgent task: ${task}\n`);

  const messages: Anthropic.MessageParam[] = [
    { role: "user", content: task },
  ];

  while (true) {
    const response = await client.messages.create({
      model: "claude-opus-4-7",
      max_tokens: 4096,
      tools,
      messages,
    });

    for (const block of response.content) {
      if (block.type === "text") {
        console.log("Agent:", block.text);
      }
    }

    if (response.stop_reason === "end_turn") break;

    if (response.stop_reason === "tool_use") {
      const toolResults: Anthropic.ToolResultBlockParam[] = [];

      for (const block of response.content) {
        if (block.type === "tool_use") {
          console.log(`\nCalling tool: ${block.name}`, block.input);
          const result = await handleToolCall(
            block.name,
            block.input as Record<string, unknown>
          );
          console.log("Tool result:", JSON.parse(result));

          toolResults.push({
            type: "tool_result",
            tool_use_id: block.id,
            content: result,
          });
        }
      }

      messages.push({ role: "assistant", content: response.content });
      messages.push({ role: "user", content: toolResults });
    }
  }
}
```

## What Happens Under the Hood

```
1. Agent calls GET /api/data
2. Server returns 402 + { amount: "0.001 USDC", address: "0x...", network: "base-sepolia" }
3. withPaymentInterceptor catches the 402
4. Interceptor signs a USDC transfer on Base Sepolia
5. Interceptor retries the request with X-Payment header
6. Server verifies payment via x402.org/facilitator
7. Server returns 200 + data
```

## Common Errors

| Error | Cause | Fix |
|---|---|---|
| `insufficient funds` | Wallet has no testnet USDC | Get from faucet.circle.com |
| `network mismatch` | Wrong chain configured | Ensure `base-sepolia` in both client and server |
| `facilitator error` | x402 facilitator unreachable | Check internet, retry |
| `gas error` | No testnet ETH for gas | Get from sepoliafaucet.com |
